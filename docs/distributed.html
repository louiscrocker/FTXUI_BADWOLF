Redis Integration

Three modes: pub/sub subscribe, key polling, and blocking queue (BLPOP). All auto-reconnecting.

Pub/Sub Subscribe

// Subscribe to a Redis channel
auto src = RedisSubscribe("my:channel");

// Or use the full API with config
RedisConfig cfg;
cfg.host = "redis.example.com";
cfg.port = 6379;
cfg.password = "secret";
auto src = RedisSource<string>::Subscribe("my:channel", cfg);

src->OnData([](const string& msg) {
    // Called on each published message
    std::cout << "Received: " << msg << "\n";
});
src->Start();

// Test with: redis-cli PUBLISH my:channel "hello world"

Key Polling

// Poll a key every 500ms, emit only on changes
auto src = RedisPollKey("app:status", chrono::milliseconds(500));

src->OnData([](const string& val) {
    std::cout << "Status changed: " << val << "\n";
});
src->Start();

// Test with: redis-cli SET app:status "running"

JSON Subscribe

// Parse messages as JSON automatically
auto src = RedisSubscribeJson("metrics:json");

src->OnData([](const JsonValue& val) {
    if (val.has("cpu"))
        std::cout << "CPU: " << val["cpu"].as_double() << "%\n";
});
src->Start();

// Test: redis-cli PUBLISH metrics:json '{"cpu":42.5,"mem":1024}'

DistributedState binding

// Bind a Redis source directly to reactive state
auto src = RedisSubscribe("app:theme");
DistributedState<string> theme("current-theme", src, "dark");

// Reactive — UI re-renders automatically when Redis publishes
theme.OnChange([](const string& t) { /* re-render */ });

// Check sync status
if (theme.IsSynced()) { /* received at least one value */ }
std::cout << "Source: " << theme.SourceName(); // redis://localhost:6379/app:theme

Kafka Integration

Native Kafka wire protocol v0 over TCP. No librdkafka required — pure POSIX sockets.

Basic Consumer

KafkaConfig cfg;
cfg.brokers = "kafka.example.com:9092";
cfg.group_id = "my-consumer-group";
cfg.reconnect_delay = chrono::milliseconds(3000);

auto src = make_shared<KafkaSource>("my-topic", cfg);

src->OnData([](const string& msg) {
    std::cout << "Kafka: " << msg << "\n";
});
src->Start();

cout << src->Name(); // kafka://kafka.example.com:9092/my-topic

Connection Status

auto src = make_shared<KafkaSource>("events");
src->Start();

// Check state at any time
bool connected = src->IsConnected();
string err = src->last_error();

// Stop gracefully
src->Stop(); // waits for worker thread to exit

gRPC Streaming

HTTP/1.1 gRPC-Web framing over plain TCP. Works with gRPC-Web proxies or grpc-gateway.

Server-Streaming RPC

GrpcStreamConfig cfg;
cfg.host = "api.example.com";
cfg.port = 8080;
cfg.service = "myapp.StreamService";
cfg.method = "LiveFeed";
cfg.request_json = R"({"filter":"all"})";
cfg.reconnect_delay = chrono::milliseconds(5000);

auto src = make_shared<GrpcStreamSource>(cfg);

src->OnData([](const JsonValue& frame) {
    if (!frame.is_null())
        std::cout << "gRPC frame: " << frame["event"].as_string() << "\n";
});
src->Start();

// Falls back gracefully if server is unavailable
// Check: src->last_error() for connection errors

Distributed Dashboard

Render multiple live sources in an auto-updating grid layout.

DistributedDashboard dashboard;

dashboard
  .AddSource("Redis Events",    RedisSubscribe("app:events"))
  .AddSource("Redis Status",    RedisPollKey("app:status"))
  .AddSource("Kafka Logs",      make_shared<KafkaSource>("logs"))
  .AddJsonSource("gRPC Metrics", make_shared<GrpcStreamSource>(grpc_cfg))
  .WithLayout(2);  // 2 columns

auto component = dashboard.Build();

// Use in your screen loop — auto-refreshes when data arrives
auto screen = ScreenInteractive::Fullscreen();
screen.Loop(component);

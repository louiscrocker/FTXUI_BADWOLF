Add FTXUI BadWolf to your CMake project

Use CMake's FetchContent to automatically download and configure FTXUI BadWolf. No manual downloads or system installs required.

# CMakeLists.txt
cmake_minimum_required(VERSION 3.14)
project(my_tui_app CXX)
set(CMAKE_CXX_STANDARD 20)

include(FetchContent)
FetchContent_Declare(ftxui
  GIT_REPOSITORY https://github.com/louiscrocker/FTXUI_BADWOLF.git
  GIT_TAG        main
  GIT_SHALLOW    TRUE
)
FetchContent_MakeAvailable(ftxui)

🎯 Single-Header (Zero Build System)

Drop dist/ftxui.hpp into your project. No cmake, no linking.

// myapp.cpp
#define BADWOLF_IMPLEMENTATION
#include "ftxui.hpp"

int main() {
  auto screen = ftxui::ScreenInteractive::TerminalOutput();
  auto console = ftxui::ui::Console::Create();
  console->Info("Hello, BadWolf!");
  screen.Loop(console->Build());
}
g++ -std=c++20 -pthread -o myapp myapp.cpp

Uses the STB-style BADWOLF_IMPLEMENTATION guard — declare it in exactly one .cpp file. All other files can #include "ftxui.hpp" without the define for declarations only.

Link the library to your target

Link ftxui::ui for the full high-level API. It transitively links ftxui::component, ftxui::dom, and ftxui::screen.

add_executable(my_tui_app main.cpp)

target_link_libraries(my_tui_app PRIVATE
    ftxui::ui       # Full high-level API (recommended)
)

# Or link individual modules if you need less:
# ftxui::component  — interactive components only
# ftxui::dom        — layout elements only
# ftxui::screen     — terminal I/O only

Hello World — 10 lines

Your first FTXUI BadWolf app. RunFullscreen takes over the terminal, renders your component, and handles the event loop.

#include <ftxui/ui/all.hpp>
using namespace ftxui::ui;

int main() {
    auto app = Layout::Vertical({
        Text("Hello from FTXUI! Press Q to quit."),
    });

    RunFullscreen(app);
    return 0;
}
Build it: cmake -B build && cmake --build build && ./build/my_tui_app

Apply a theme

One line sets the color palette for every component in your app. Call it before creating any components.

#include <ftxui/ui/all.hpp>
using namespace ftxui::ui;

int main() {
    SetTheme(Theme::Nord());   // Arctic Nord palette
    // or: Dark(), Light(), Dracula(), Monokai()

    auto app = Layout::Vertical({
        Text("Themed app!"),
    });
    RunFullscreen(app);
}

Create a form

Collect user input with automatic tab-navigation, validation, and submit/cancel callbacks.

std::string name, email;

auto form = Form()
    .Title("User Registration")
    .Field("Full Name", name)
    .Field("Email", email)
    .OnSubmit([&]{
        Notify("Welcome, " + name + "!", Success);
    })
    .OnCancel([&]{ Navigate("/"); })
    .Build();

RunFullscreen(form);

Add a chart

LineChart renders in high-resolution braille. Use RenderFlex() to fill all available space.

std::vector<double> cpu_data = {20,45,38,67,89,55,42};

auto chart_elem = LineChart()
    .Title("CPU Usage %")
    .Series("cpu", cpu_data, Color::Red())
    .XLabels({"Mon","Tue","Wed","Thu","Fri","Sat","Sun"})
    .RenderFlex();   // fills all available space

// Wrap as a Component to put in Layout:
auto chart_comp = Renderer([=]{ return chart_elem; });

Add a data table

Generic over any struct. Define columns with lambdas — no code generation or reflection required.

struct Server {
    std::string host, status;
    int port;
    float latency;
};

std::vector<Server> servers = {
    {"web-01", "UP",   80,   2.1f},
    {"web-02", "UP",   80,   3.7f},
    {"db-01",  "DOWN", 5432, 0.0f},
};

auto table = DataTable<Server>()
    .Columns({
        {"Host",    [](auto& s){ return s.host; }},
        {"Status",  [](auto& s){ return s.status; }},
        {"Port",    [](auto& s){ return std::to_string(s.port); }},
        {"Latency", [](auto& s){ return fmt("{:.1f}ms", s.latency); }},
    })
    .Rows(servers)
    .Build();

Use Reactive state

Share state between components and threads. The UI refreshes automatically when values change.

auto count = std::make_shared<Reactive<int>>(0);

auto display = Renderer([count]{
    return text("Count: " + std::to_string(count->Get()));
});

auto button = Button("Increment", [count]{
    count->Set(count->Get() + 1);
});

// count->Set() from any thread also triggers re-render

Add a world map

The built-in GeoJSON contains the full NE110m Natural Earth dataset. No external files needed.

auto map = GeoMap()
    .Data(WorldMapGeoJSON())  // built-in dataset
    .Center(0.0, 20.0)
    .Zoom(1.0)
    .Overlay({"HQ", -74.0, 40.7, "★"})
    .OnSelect([&](std::string country) {
        Notify(country, Info);
    })
    .Build();

Run your app

Three run modes: fullscreen (takes over terminal), fit-content (shrinks to content size), or loop (manual event control).

// Option 1: Take over the full terminal
RunFullscreen(app);

// Option 2: Fit terminal to content size
RunFitContent(app);

// Option 3: Manual loop (advanced)
RunLoop(app);

// Option 4: With debug overlay (Ctrl+D toggles)
RunFullscreen(WithDebugOverlay(app));
Complete Example

Your First Full App (~50 lines)

Combines a theme, form, data table, chart, and notification in one cohesive app.

#include <ftxui/ui/all.hpp>
using namespace ftxui::ui;
using namespace ftxui;

int main() {
    // 1. Theme
    SetTheme(Theme::Nord());

    // 2. State
    std::string search_query;
    struct User { std::string name, role; bool active; };
    std::vector<User> users = {
        {"alice", "admin", true},
        {"bob",   "user",  true},
        {"carol", "user",  false},
    };

    // 3. Chart data
    std::vector<double> logins = {12,34,28,56,67,89,45};

    // 4. Components
    auto header = Renderer([]{ return text("  User Dashboard  ") | bold | center; });

    auto table = DataTable<User>()
        .Columns({
            {"Name",   [](auto& u){ return u.name; }},
            {"Role",   [](auto& u){ return u.role; }},
            {"Active", [](auto& u){ return u.active ? "✓" : "✗"; }},
        })
        .Rows(users)
        .OnSelect([&](User u){
            Notify("Selected: " + u.name, Info);
        })
        .Build();

    auto chart_comp = Renderer([&]{
        return LineChart()
            .Title("Logins / Day")
            .Series("logins", logins, Color::Green())
            .RenderFlex();
    });

    // 5. Layout
    auto app = Layout::Vertical({
        header,
        Layout::Horizontal({ table, chart_comp }),
    });

    // 6. Launch
    RunFullscreen(WithDebugOverlay(app));
    return 0;
}

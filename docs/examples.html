Build Any Example in 60 Seconds

# Clone and build all examples
git clone https://github.com/louiscrocker/FTXUI_BADWOLF.git
cd FTXUI

cmake -S . -B build \
    -DCMAKE_BUILD_TYPE=Release \
    -DFTXUI_BUILD_EXAMPLES=ON

cmake --build build --parallel $(nproc)

# Run any example
./build/examples/component/globe_demo
./build/examples/component/world_ops_demo
./build/examples/component/showcase

# Or build a specific example only
cmake --build build --target ftxui_example_globe_demo

TypewriterText

Returns a Component that renders text one character at a time, driven by the global AnimationLoop. 

#include "ftxui/ui/typewriter.hpp"

using namespace ftxui::ui;

// Basic usage
auto tw = TypewriterText("Hello, World!");

// Custom config
TypewriterConfig cfg;
cfg.chars_per_second = 50;
cfg.show_cursor      = true;
cfg.cursor_char      = '|';
cfg.cursor_blink     = true;
cfg.blink_hz         = 2;
cfg.on_complete      = [] { std::cout << "Done!\n"; };
cfg.on_char          = [](char c) { /* fired per char */ };

auto tw = TypewriterText("FTXUI BadWolf ready.", cfg);
ftxui::ui::RunFullscreen(tw);

TypewriterConfig Reference

Field Type Default Description
chars_per_second int 30 Typing speed. 0 = instant display.
show_cursor bool true Show cursor glyph while typing.
cursor_char char '_' The cursor glyph (e.g. '_', '|', '█').
cursor_blink bool true Animate cursor visibility.
blink_hz int 2 Cursor blink rate in Hz.
on_complete function<void()> null Fires once when all chars displayed.
on_char function<void(char)> null Fires for each character displayed.

TypewriterSequence

Chain multiple animations using a fluent builder. Steps execute in order with optional pauses and clears. 

TypewriterConfig matrix;
matrix.chars_per_second = 20;

auto seq = TypewriterSequence()
    .Then("INITIATING BADWOLF SYSTEMS...", matrix)
    .Pause(std::chrono::milliseconds(600))
    .Then("\nLOADING MODULES...", matrix)
    .Pause(std::chrono::milliseconds(400))
    .Clear()
    .Then("ALL SYSTEMS NOMINAL.", matrix)
    .OnComplete([] { /* sequence finished */ })
    .Build();

ftxui::ui::RunFullscreen(seq);

Console

A thread-safe scrollable output console with styled log levels and animated typewriter output. 

auto con = Console::Create(500);  // max 500 lines

// Styled log helpers
con->Info("Server started on port 8080");     // cyan
con->Warn("Memory usage at 87%");             // yellow
con->Error("Connection pool exhausted");      // red bold
con->Success("Task completed in 42ms");       // green

// Plain output
con->PrintLine("Hello, World!", Color::White);
con->PrintBold("=== Section Header ===");

// Animated typewriter output
TypewriterConfig tw;
tw.chars_per_second = 40;
con->TypePrint("Streaming 50,000 rows...", tw, Color::GreenLight);

// Clear all lines
con->Clear();

// Build and run
auto comp = con->Build(" Application Log ");
ftxui::ui::RunFullscreen(comp);

ConsolePrompt — Getting Started

A full REPL component combining a scrollable console output with an interactive input line. Supports command history via ↑/↓ arrows, customizable prompt, and typewriter responses. 

auto con = Console::Create();
con->PrintBold("Welcome to MyApp REPL v1.0");
con->Info("Type 'help' to get started.");

ConsolePromptConfig cfg;
cfg.prompt_str        = "myapp> ";
cfg.prompt_color      = Color::GreenLight;
cfg.placeholder       = "enter command...";
cfg.history_size      = 100;
cfg.typewriter_output = true;

cfg.handler = [](const std::string& cmd) -> std::string {
    if (cmd == "help")    return "Commands: help quit version";
    if (cmd == "version") return "myapp v1.0.0";
    if (cmd == "quit") {
        if (App* a = App::Active()) a->Exit();
        return "";
    }
    return "[ERROR] Unknown: " + cmd;
};

auto prompt = ConsolePrompt(con, cfg);
ftxui::ui::RunFullscreen(prompt);

Example: Star Wars Opening Crawl

Recreate the iconic Star Wars scrolling text using TypewriterSequence with yellow color and dramatic pauses. 

TypewriterConfig crawl;
crawl.chars_per_second = 15;
crawl.show_cursor      = false;

auto comp = TypewriterSequence()
    .Then("A long time ago in a galaxy far,\nfar away....", crawl)
    .Pause(std::chrono::milliseconds(1000))
    .Clear()
    .Then("STAR\nWARS", crawl)
    .Pause(std::chrono::milliseconds(500))
    .Then("\n\nEpisode IV\nA NEW HOPE\n", crawl)
    .Pause(std::chrono::milliseconds(300))
    .Then("It is a period of civil war.\n"
          "Rebel spaceships, striking\n"
          "from a hidden base...", crawl)
    .OnComplete([] {
        if (App* a = App::Active()) a->Exit();
    })
    .Build();

// Render in yellow — the classic crawl color
auto wrapped = Renderer(comp, [comp]() -> Element {
    return comp->Render() | color(Color::Yellow) | hcenter;
});

ftxui::ui::RunFullscreen(wrapped);

WithTypewriter — Intro Animations

Wrap any component with a typewriter intro. The intro plays first; when complete, the real component is revealed. 

auto my_dashboard = BuildDashboard();

TypewriterConfig intro;
intro.chars_per_second = 40;

auto app = WithTypewriter(
    my_dashboard,
    "BADWOLF OS v1.0  INITIALIZING...",
    intro);

ftxui::ui::RunFullscreen(app);

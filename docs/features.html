Layout & Structure

Layout

Layout::Vertical / Horizontal / Grid

Stack components vertically, horizontally, or in a responsive grid. All layouts participate in the flexbox-inspired size negotiation system.

auto app = Layout::Vertical({
    Text("Top panel"),
    Layout::Horizontal({
        SideBar(),
        MainContent(),
    }),
    StatusBar(),
});

// Grid: 3 columns
auto grid = Layout::Grid(3, {
    Card("A"), Card("B"), Card("C"),
    Card("D"), Card("E"), Card("F"),
});

Form

Declarative form builder with automatic layout, validation, tab navigation between fields, and submit/cancel callbacks.

std::string name, email, pass;
auto form = Form()
    .Title("Create Account")
    .Field("Name", name)
    .Field("Email", email)
    .Password("Password", pass)
    .OnSubmit([&]{ createUser(name, email); })
    .OnCancel([&]{ Navigate("/login"); })
    .Build();

Router

Client-side routing for multi-page terminal apps. Navigate between views without re-initializing the app loop. Lazy component factory functions.

auto app = Router()
    .Route("/",        []{ return DashboardView(); })
    .Route("/users",   []{ return UsersView(); })
    .Route("/settings",[]{ return SettingsView(); })
    .Default("/")
    .Build();

// Navigate from anywhere:
Navigate("/users");

DataTable<T>

Generic data table with custom column renderers, row selection, keyboard navigation, and mouse click callbacks. No boilerplate.

struct Process { int pid; std::string name; float cpu; };

auto table = DataTable<Process>()
    .Columns({
        {"PID",  [](auto& p){ return std::to_string(p.pid); }},
        {"Name", [](auto& p){ return p.name; }},
        {"CPU%", [](auto& p){ return fmt("{:.1f}", p.cpu); }},
    })
    .Rows(processes)
    .OnSelect([&](Process p){ showDetail(p); })
    .Build();

VirtualList<T>

O(visible) rendering — renders only what fits on screen. 50,000 rows scroll as fast as 50. Custom row renderer with selected state.

auto list = VirtualList<LogEntry>()
    .Items(all_50k_logs)
    .Renderer([](const LogEntry& e, bool sel) {
        auto color = e.level == "ERROR"
            ? Color::Red() : Color::Default();
        return hbox({
            text(e.timestamp) | dim,
            text(e.message) | color(color),
        });
    })
    .OnSelect([&](LogEntry e){ showDetail(e); })
    .Build();

LineChart

Multi-series line chart with braille rendering at 8× terminal resolution. Auto-scales Y axis, supports X labels, legend, title, and RenderFlex() to fill any space.

std::vector<double> cpu = {20,45,30,67,89,55,42};
std::vector<double> mem = {60,62,61,64,70,68,69};

auto chart = LineChart()
    .Title("System Metrics")
    .Series("CPU", cpu, Color::Red())
    .Series("Memory", mem, Color::Blue())
    .XLabels({"Mon","Tue","Wed","Thu","Fri","Sat","Sun"})
    .RenderFlex();  // fills all available space

Category 05

Reactive State Management

State

Reactive<T> / Computed<T>

Thread-safe observable values. Set from any thread — the UI refreshes automatically. Computed<T> derives new values reactively. ReactiveGroup batches updates.

auto count = std::make_shared<Reactive<int>>(0);

// Derived value — auto-updates when count changes
auto doubled = MakeComputed(count,
    [](int v){ return v * 2; });

// From background thread:
count->Set(count->Get() + 1); // triggers UI refresh

// Batch multiple updates (one refresh):
ReactiveGroup g;
g.Batch([&]{
    x->Set(10); y->Set(20); z->Set(30);
});
